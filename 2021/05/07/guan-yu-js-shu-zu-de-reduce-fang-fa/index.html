关于JS数组的reduce方法

FrontEnd
  |  
 阅读次数

语法

arr.reduce(callback(accumulator, currentValue[, index[, array]])[, initialValue])

解析

回调函数有四个参数

  1. accumulator:累加器
  2. currentValue:当前值
  3. index(可选):当前索引
  4. array(可选):调用reduce的数组

    第二个参数(可选)

    initialValue:初始值,作为第一次调用 callback函数时的第一个参数的值。 如果没有提供初始值，则将使用数组中的第一个元素。 在没有初始值的空数组上调用 reduce 将报错。 (摘自MDN文档)

    以下是几个案例

    Case 1

    var sum = [0, 1, 2, 3, 4, 5].reduce((accumulator, currentValue, currentIndex, array) => {
        console.log("currentValue: " + accumulator);
        return accumulator + currentValue
    }, 10)
    /*currentValue: 10
    currentValue: 10
    currentValue: 11
    currentValue: 13
    currentValue: 16
    currentValue: 20 */

console.log(sum) //25
    通过这个案例了解基本执行顺序

    Case 2

    var flattend = [
    [0, 1],
    [2, 3],
    [4, 5]
].reduce(
    function(a, b) {
        return a.concat(b)
    }
)

console.log(flattend)
    也可通过此函数对二维数组进行铺平,起到flat/flatMap的作用

    Case 3

    const names = ['ly', 'lly', 'lyy', 'ly', 'lh']

var countName = names.reduce(
    function(allName, name) {
        if (name in allName) {
            allName[name]++
        } else {
            allName[name] = 1
        }
        return allName
    }, {}
)
console.log(countName)
//{ ly: 2, lly: 1, lyy: 1, lh: 1 }
    统计数组中每个元素出现的次数,类似上面

    Case 4

    var people = [
    { name: 'ly', age: 20 },
    { name: 'lg', age: 24 },
    { name: 'lhc', age: 20 }
]

function groupBy(Array, Property) {
    return Array.reduce((acc, obj) => {
        var key = obj[Property]
        if (!acc[key]) {
            acc[key] = []
        }
        acc[key].push(obj)
        return acc
    }, {})
}

var groupByAgePeople = groupBy(people, 'age') //第二个参数必须是字符串类型
console.log(groupByAgePeople);
/*
{
  '20': [ { name: 'ly', age: 20 }, { name: 'lhc', age: 20 } ],
  '24': [ { name: 'lg', age: 24 } ]
}
*/
    按照属性对对象分类,也是类似上面…

    Case 5

    function RunPromiseInSequence(arr, input) {
    return arr.reduce((PromiseChain, currentFunc) =>
        PromiseChain.then(currentFunc), Promise.resolve(input))
}

function punc1(a) {
    return new Promise((resolve, reject) =>
        resolve(a * 3)
    )
}

function punc2(a) {
    return new Promise((resolve, reject) =>
        resolve(a * 2)
    )
}

function func3(a) {
    return a * 5
}

function punc4(a) {
    return new Promise((resolve, reject) =>
        resolve(a * 4)
    )
}

const Parr = [punc1, punc2, func3, punc4]

RunPromiseInSequence(Parr, 2)
    .then(console.log) //240
    按顺序执行promise对象
    重点说这个,先定义一个函数RunPromiseInSequence,第一个参数为一个数组arr,对这个数组使用reduce方法,简单来说,这个数组里面每个元素都是一个函数,PromiseChain相当于accumulator,只不过不再是累加,而是依次调用函数,每个函数中又对它的参数解析,按数组顺序依次执行。对于第三个函数,它会被.then包裹在一个已经解决的Promise对象中。(说真的,我不懂这句话什么意思,我感觉应该是Promise.resolve(input)把上一步的输出(一个普通Number),会再解析为一个resolved的Promise对象?)
